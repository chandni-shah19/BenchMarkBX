001package org.benchmarx.examples.familiestopersons.testsuite.level1.incr.wocorrs.delta.auto;
002
003import org.benchmarx.BXTool;
004import org.benchmarx.examples.familiestopersons.testsuite.Decisions;
005import org.benchmarx.examples.familiestopersons.testsuite.FamiliesToPersonsTestCase;
006import org.junit.Test;
007
008import Families.FamilyRegister;
009import Persons.PersonRegister;
010
011public class RenamingPerson extends FamiliesToPersonsTestCase {
012        
013        public RenamingPerson(BXTool<FamilyRegister, PersonRegister, Decisions> tool) {
014                super(tool);
015        }
016
017        /**
018         * <b>Test</b> for changing a person's first name.
019         * <p>
020         * <b>Expected</b>: only the first name of the corresponding family member
021         * should be changed.
022         * <p>
023         * <b>Classification</b>: incr-wocorr-delta-auto
024         * <ul>
025         * <li><b>incr</b>: changing a persons first name requires old consistent
026         * family model as the information if females are daughters or mothers (and
027         * males analogously) would otherwise be lost.
028         * <li><b>wocorr</b>: assumption of unique names can be used here to compute
029         * correspondences correctly.
030         * <li><b>delta</b>: renaming cannot be distinguished from combined deletion
031         * and creation.
032         * <li><b>auto</b>: propagation is deterministic.
033         * </ul>
034         */
035        @Test
036        public void testFirstNameChangePerson() {
037                tool.initiateSynchronisationDialogue();
038                tool.performAndPropagateSourceEdit(util
039                                .execute(helperFamily::createSimpsonFamily)
040                                .andThen(helperFamily::createFatherHomer)
041                                .andThen(helperFamily::createSimpsonFamilyMembers));
042                        
043                //----------------
044                tool.performAndPropagateTargetEdit(helperPerson::firstNameChangeOfHomer);
045                //----------------
046                
047                util.assertTarget("PersonNameChange");
048                util.assertSource("MemberNameChange");
049        }
050}
