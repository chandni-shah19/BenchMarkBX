001package org.benchmarx.examples.familiestopersons.testsuite.level1.incr.wcorrs.delta.auto;
002
003import org.benchmarx.BXTool;
004import org.benchmarx.examples.familiestopersons.testsuite.Decisions;
005import org.benchmarx.examples.familiestopersons.testsuite.FamiliesToPersonsTestCase;
006import org.junit.Test;
007
008import Families.FamilyRegister;
009import Persons.PersonRegister;
010
011public class RenamingPersons extends FamiliesToPersonsTestCase {
012        
013        public RenamingPersons(BXTool<FamilyRegister, PersonRegister, Decisions> tool) {
014                super(tool);
015        }
016        
017        /**
018         * <b>Test</b> for changing a person's first name (where another person with
019         * the same name exists).
020         * <p>
021         * <b>Expected</b>: the first name of the corresponding member in the
022         * families model should be changed.
023         * <p>
024         * <b>Classification</b>: incr-wcorr-delta-auto
025         * <ul>
026         * <li><b>incr</b>: the old family register is required as it is impossible
027         * to guess if female/male persons are mothers/fathers or daughters/sons in
028         * the family register.
029         * <li><b>wcorr</b>: it's impossible to guess which family member has to be
030         * renamed in the families model as two persons (one of which is to be
031         * renamed) have the same name.
032         * <li><b>delta</b>: renaming is inherently delta-based as it cannot be
033         * distinguished from combined deletion and creation.
034         * <li><b>auto</b>: propagation is deterministic so no choice required.
035         * </ul>
036         */
037        @Test
038        public void testFirstNameChangeOfNonUniquePerson() {
039                tool.initiateSynchronisationDialogue();
040                tool.performAndPropagateSourceEdit(util
041                                .execute(helperFamily::createSimpsonFamily)
042                                .andThen(helperFamily::createFatherHomer)
043                                .andThen(helperFamily::createSimpsonFamilyMembers));
044                tool.performAndPropagateTargetEdit(helperPerson::createOtherBart);
045                        
046                //----------------
047                tool.performAndPropagateTargetEdit(helperPerson::firstNameChangeOfBart);
048                //----------------
049                
050                util.assertTarget("PersonNameChangeOther");
051                util.assertSource("MemberNameChangeOther");
052        }
053}
