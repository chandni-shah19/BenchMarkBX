001package org.benchmarx.examples.familiestopersons.testsuite.level1.batch.wocorrs.state.auto;
002
003import org.benchmarx.BXTool;
004import org.benchmarx.examples.familiestopersons.testsuite.Decisions;
005import org.benchmarx.examples.familiestopersons.testsuite.FamiliesToPersonsTestCase;
006import org.junit.Test;
007
008import Families.FamilyRegister;
009import Persons.PersonRegister;
010
011public class Initialisation extends FamiliesToPersonsTestCase {
012        
013        public Initialisation(BXTool<FamilyRegister, PersonRegister, Decisions> tool) {
014                super(tool);
015        }
016
017        /**
018         * <b>Test</b> for agreed upon starting state.
019         * <p>
020         * <b>Expect</b> root elements of both source and target models.
021         * <p>
022         * <b>Classification</b>: batch-wocorr-state-auto
023         * <ul>
024         * <li><b>batch</b>: this is the initial state for synchronisation, so an
025         * old(er) consistent state doesn't exist.
026         * <li><b>wocorr</b>: no need for traceability links (in general, wcorr and
027         * batch does not make sense).
028         * <li><b>state</b>: easy to guess/compute the delta involved here (creating
029         * a root container).
030         * <li><b>auto</b>: propagation is deterministic so no choice involved.
031         * <ul>
032         */
033        @Test
034        public void testInitialiseSynchronisation()
035        {
036                //------------
037                tool.initiateSynchronisationDialogue();
038                //------------
039                
040                util.assertSource("RootElementFamilies");
041                util.assertTarget("RootElementPersons");
042        }
043        
044        /**
045         * <b>Test</b> for name change of an empty family, i.e, a family without any family members.
046         * <p>
047         * <b>Expect</b> no change in the persons model.
048         * <p>
049         * <b>Classification</b>: batch-wocorr-state-auto
050         * <ul>
051         * <li><b>batch</b>: person register can be recreated from scratch without loss of information.
052         * <li><b>wocorr</b>: no traceability links required.
053         * <li><b>state</b>: in this case, renaming and delete+create have the same effect.
054         * <li><b>auto</b>: propagation is deterministic so no choice involved.
055         * </ul>
056         */
057        @Test
058        public void testFamilyNameChangeOfEmpty()
059        {
060                tool.initiateSynchronisationDialogue();
061                tool.performAndPropagateSourceEdit(helperFamily::createSimpsonFamily);
062                
063                //------------
064                tool.performAndPropagateSourceEdit(helperFamily::familyNameSimpsonChangeEmpty);
065                //------------
066                                
067                util.assertSource("NameChangeFamilyEmpty");
068                util.assertTarget("NameChangePersonEmpty");
069        }
070}
