001package org.benchmarx.examples.familiestopersons.testsuite.level3_n.incr.wocorrs.state.auto;
002
003import org.benchmarx.BXTool;
004import org.benchmarx.examples.familiestopersons.testsuite.Decisions;
005import org.benchmarx.examples.familiestopersons.testsuite.FamiliesToPersonsTestCase;
006import org.junit.Test;
007
008import Families.FamilyRegister;
009import Persons.PersonRegister;
010
011public class CreateMultiFamilyMembers extends FamiliesToPersonsTestCase {
012        
013        public CreateMultiFamilyMembers(BXTool<FamilyRegister, PersonRegister, Decisions> tool) {
014                super(tool);
015        }
016
017        /**
018         * <b>Test</b> for creation of multiple family members in an existing
019         * family.
020         * <p>
021         * <b>Expect</b> the creation of multiple persons (male/female as required)
022         * associated to the new family members.
023         * <p>
024         * <b>Classification</b>: incr-wocorr-state-auto
025         * <ul>
026         * <li><b>incr</b>: creation of family and multiple family member requires
027         * old(er) states, otherwise birthdays would be lost (here for the father).
028         * <li><b>wocorr</b>: assumption based on unique names works for this
029         * example.
030         * <li><b>state</b>: easy to guess/compute the delta involved here based on
031         * the old and new states (what was added is clear).
032         * <li><b>auto</b>: propagation is deterministic so no choice involved.
033         * <ul>
034         */
035        @Test
036        public void testCreateMultiFamilyMember()
037        {
038                tool.initiateSynchronisationDialogue();
039                tool.performAndPropagateSourceEdit(helperFamily::createSimpsonFamily);
040                tool.performAndPropagateSourceEdit(helperFamily::createFatherHomer);
041                
042                //------------
043                tool.performAndPropagateSourceEdit(helperFamily::createSimpsonFamilyMembers);
044                //------------
045                
046                util.assertSource("FamilyWithMultiFamilyMember");
047                util.assertTarget("PersonWithMultiMember");
048        }
049}
