005
006import org.benchmarx.BXTool;

023@RunWith(Parameterized.class)
024public abstract class FamiliesToPersonsTestCase {
025
026        protected BXTool<FamilyRegister, PersonRegister, Decisions> tool;
027        protected Comparator<FamilyRegister> familiesComparator;
028        protected Comparator<PersonRegister> personsComparator;
029        protected BenchmarxUtil<FamilyRegister, PersonRegister, Decisions> util;
030        protected FamilyHelper helperFamily;
031        protected PersonHelper helperPerson;
032
033        @Before
034        public void initialise() {
035                familiesComparator = new FamiliesComparator();
036                personsComparator = new PersonsComparator();
037                util = new BenchmarxUtil<>(familiesComparator, personsComparator, tool);
038                helperFamily = new FamilyHelper();
039                helperPerson = new PersonHelper();
040        }
041
042        @Parameters
043        public static Collection<BXTool<FamilyRegister, PersonRegister, Decisions>> tools() {
044                return Arrays.asList(
045                                new EMoflonFamiliesToPersons()
046                                ,
047                                new MediniQVTR()
048                        );
049        }
050        
051        protected FamiliesToPersonsTestCase(BXTool<FamilyRegister, PersonRegister, Decisions> tool) {
052                this.tool = tool; 
053        }
054}
