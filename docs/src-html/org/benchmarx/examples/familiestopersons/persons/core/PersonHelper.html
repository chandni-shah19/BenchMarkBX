017        private Person getFromRegister(String name, PersonRegister register) {
018                Optional<Person> person = register.getPersons().stream()
019                                .filter(p -> p.getName().equals(name))
020                                .findAny();
021                                
022                assertTrue(person.isPresent());
023                return person.get();
024        }

028                person.setName("Simpson, Homer");
029                register.getPersons().add(person);
030        }

032        public void birthdayChangeOfHomer(PersonRegister register) {
033                Person person = getFromRegister("Simpson, Homer", register);
034                
035                Calendar cal = Calendar.getInstance();
036                cal.set(2013, Calendar.JANUARY, 9, 10, 11, 12); 
037                Date date = cal.getTime();
038                person.setBirthday(date);
039        }

058        public void firstNameChangeOfBart(PersonRegister register) {
059                Calendar cal = Calendar.getInstance();
060                cal.set(2013, Calendar.JANUARY, 9, 10, 11, 12); 
061                Date date = cal.getTime();
062                
063                for(int i =0;i<register.getPersons().size();i++)
064                {
065                                if(register.getPersons().get(i).getBirthday().toString().equals(date.toString()))
066                                {
067                                        register.getPersons().get(i).setName("Simpson, BartX");
068                                }
069                }
070                
071        }
